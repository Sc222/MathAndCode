Python Математическое представление 

def count(n):
    if n > 2:
        return count(n - 1) + count(n - 2)
    else:
        return 1


def output(n):
    if n > 2:
        print("*")
        return output(n - 1)
    else:
        return 1
                    

count(n) = count(n - 1) + count(n - 2) при n > 2;
count(n) = 1 при всех остальных n;

output(n) = print("*") + output(n - 1) при n > 2;
output(n) = 1 при всех остальных n;
